let inpNum;
let count = 0;
do {
  inpNum = window.prompt("숫자를 입력하세요");
  count++;
} while (inpNum === "" || isNaN(inpNum));

if (inpNum !== null) {
  console.log(`입력된 숫자는 ${inpNum}`);
} else {
  console.log("취소를 누르셨네요...");
}
