object type


const object = {
  name: "dayoung",
  "name-en": "dianne",
  sayHello: function () {
    console.log("Hello!");
  },
  sayHi() {
    console.log("Hi!");
  },
};

console.log(`object["name"] :`, object["name"]);
console.log(`object.name :`, object.name);
console.log(`object.age:`, object.age);

console.log(`- object["sayHello"]();`);
object["sayHello"]();
console.log(`- object.sayHi();`);
object.sayHi();

object["name"] = "다람이";
console.log(`변경) object.name :`, object.name);

object.MBTI = "ENFJ";
console.log(`추가) object.MBTI :`, object.MBTI);

delete object["MBTI"];
console.log(`삭제) object["MBTI"] :`, object["MBTI"]);
