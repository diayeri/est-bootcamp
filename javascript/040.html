(for, while) - break, continue


// for - break
for (let i = 0; i <= 5; i++) {
  if (i === 3) break;

  console.log(i);
}


// while - break
let num = 0;
while (num < 11) {
  if (num === 5) break;

  console.log(num);
  num++;
}


// for - continue
for (let i = 0; i <= 5; i++) {
  if (i === 3) continue;

  console.log(i);
}


// while - continue
num = 0;
while (num < 11) {
  num++; // 증감식이 continue 보다 위에 위치해야 무한루프가 안생김
  if (num === 5) continue;
  console.log(num);
}


// do while - continue
num = 0;
do {
  num++;
  if (num === 5) continue;
  console.log(num);
} while (num < 11);
