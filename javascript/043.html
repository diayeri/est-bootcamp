매개변수, 인자


// 함수의 매개변수(parameter), 인자(argument)
// 매개변수 기본값 설정 가능
function sum(para1 = "default1", para2) { 
  return para1 + para2;
}
sum("argu1", "argu2"); 


// 예) 매개변수와 인자의 수가 다를 때, 인자가 null, undefined 일 때
function sumNumbs(a = 10, b = 20, c = 30) {
  console.log(`-- ${a} + ${b} + ${c}`);
  return a + b + c;
}
console.log(sumNumbs(10, 20, 30, 40));
console.log(sumNumbs(10, 20));
console.log(sumNumbs(0, undefined, 0));
console.log(sumNumbs(null, 0, 0));


// 원시 call - 인자가 원시타입일 때의 호출
let number = 10;
function test(num) {
  num = num + 10;
  console.log("function", num);
}
test(number);
console.log(number);


// 객체 call - 인자가 객체타입일 때의 호출 ✅
const object = { a: 1, b: 2 };
function test2(obj) {
  obj.c = 3; // 객체가 수정됨
  obj = { name: "hati" }; // 다른 객체를 할당함
  console.log("function", obj);
}
test2(object);
console.log(object);
