반환값


function sum(a, b) {
  return a + b;
  console.log("return 문 뒤에 오는 코드 -> 실행 안됨");
}
console.log(sum(2, 3));


// return 생략시, 함수의 반환값은 없다.
function sub(a, b) {
  console.log(a - b);
}
console.log(sub(5, 2));


// if, return 을 이용한 예외처리
// return을 만나면 "함수"는 종료된다 (if는 함수가 아니고 실행 구문!)
function sumNumb(a, b) {
  if (isNaN(a) || isNaN(b)) {
    console.log("숫자가 아닙니다!");
    return;
  }
  return a + b;
}
console.log(sumNumb(10, 20));
console.log(sumNumb("hello", 2));
