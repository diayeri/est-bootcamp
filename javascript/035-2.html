반복문 예제 다르게 구현하기 - reduce, forEach, for ... of


// 2. 반의 평균을 구하는 반복문 코드 (reduce, forEach, for of 사용 X)
const score = [84, 92, 39, 55, 71];
let scoreSum = 0;
for (let i = 0; i < score.length; i++) {
  scoreSum += score[i];
}
let average = scoreSum / score.length;
console.log("average:", +average.toFixed(2));


// 3. reduce, forEach, for of 사용해서 만들어보기!
const toAverage = function (sum, length) {
  return (sum / length).toFixed(2);
};


// 3-1. reduce
let reduceSum = score.reduce(function (a, c) {
  return a + c;
}, 0);
console.log("reduce average:", (reduceSum / score.length).toFixed(2));
console.log("reduce average:", toAverage(reduceSum, score.length));


// 3-2. forEach
let forEachSum = 0;
score.forEach(function (item) {
  forEachSum += item;
});
console.log("forEach average:", (forEachSum / score.length).toFixed(2));
console.log("forEach average:", toAverage(reduceSum, score.length));


// 3-3. for of
let forOfSum = 0;
for (item of score) {
  forOfSum += item;
}
console.log("forOf average:", toAverage(forOfSum, score.length));
