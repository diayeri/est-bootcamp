switch


switch(표현식){
  case 값1:
    // 값1에 대한 실행 코드
    break;
  case 값2:
    // 값2에 대한 실행 코드
    break;
  default:
    // 모든 Case에 일치하지 않는 경우 실행할 코드
}


const today = new Date();
console.log("오늘은 : ", today.getDay());

switch (today.getDay()) {
  case 1:
    console.log("워어어어얼");
    break;
  case 2:
    console.log("화아아아아");
    break;
  case 3:
    console.log("수우우");
    break;
  case 4:
    console.log("모옥");
    break;
  case 5:
    console.log("금");
    break;
  case 6:
  case 0:
    console.log("퇼");
    break;
  default:
    console.log("무슨 요일인지 모르겠습니다");
}
