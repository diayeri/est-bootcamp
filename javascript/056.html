재귀 함수


// ## 기본형태 - 팩토리얼
function factorial(n) {
  if (n <= 1) {
    return n;
  }
  return n * factorial(n - 1);
}


// ## 반복문과 비교 - 1~5 합 구하기
// ### 반복문
let sum = 0;
for (let index = 0; index < 6; index++) {
  sum += index;
}

// ### 재귀함수
function sum(n) {
  if (n <= 1) {
    return n;
  }
  return n + sum(n - 1);
}


// 예1) 반복문으로 1~n 합 구하기
let sum1 = 0;
for (i = 0; i <= 5; i++) {
  sum1 += i;
}
console.log(sum1);


// 예2) 재귀로 1~n 합 구하기
function sum(n) {
  if (n <= 1) {
    return n;
  }
  // 5일경우에는 5 + 4~1까지 더한값
  // 4일경우에는 4 + 3~1까지 더한값
  return n + sum(n - 1);
}


// 예3) 피보나치 for문
function fibonacci(n) {
  let a = 1,
    b = 0,
    temp;
  while (n >= 0) {
    temp = a;
    a = a + b;
    b = temp;
    n--;
  }
  return b;
}
